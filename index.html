                        {name : '下拉选项4',id : 'day4'}
                 ],
                defaultValue : "day4",
                valueId : "signId",

        });

            var name=$("#selectBT1").attr("txt");
            var val=$("#selectBT1").attr("val");
            alert(name+"+"+val);
        });

2、禁用部分下拉选项

        $("#selectBT2").selects({
            data:[{'name':'下拉选项1','id':'day1'},
                  {'name':'下拉选项2','id':'day2'},
                  {'name':'下拉选项3','id':'day3'},
                  {'name':'下拉选项4','id':'day4'},
                  {'name':'下拉选项5禁用','id':'day5','disable':true},
                  {'name':'下拉选项6禁用','id':'hour6','disable':true}],
            top:30,
            left:0,
            valueField : 'id',
            textField : 'name',
            defaultName:"请选择下拉",
            defaultValue:"",
            defaultLine:6,
            className:"select_value",				
            valueId:"signId",
            zindex:19,
            isSearch:false,
            callback:function(txt,val){
            	$("#info2").html("您选择了："+txt);
            }
        });

8、参数说明

参数名描述用法
data 下拉的数据参数, 必填, 数据的名称name 和id为默认名称，可修改，修改后参数 valueField,textField做相应修改即可。disable:数据中控制下拉是否禁用某下拉列，值为：true/false 数据格式为数组:[{'name':'下拉选项1','id':'day1}]
valueField 数据中id的字段名称 valueField: 'id',//数据字段id
textField 数据中name的字段名称 textField: 'name',//数据字段名称
className 样式名称，默认class为：select_value className: "select_value",//样式
valueId 隐藏input的id名，用于记录选择当前下拉的值。可得到其值用于提交数据。 设置 valueId: "idval", 得到选择的下拉的值为 $("#idval").val()
defaultName 下拉上面默认显示的名称 如：请选择 等等。defaultValue设置时defaultName无效。 defaultName："请选择"
defaultValue 隐藏input的默认值，设置默认值会自动选择该条数据，不设置默认显示 defaultName 的名称见例1 defaultValue：2
defaultLine 默认显示数据条数，超过这个条数出滚动条，见例4 defaultLine: 5,//默认下拉显示条数
zindex 设置下拉的z-index值 zindex: '1',
firstnum 添加下拉中的第一条数名称 默认值为空 firstnum:"第一条数据"
isSearch 是否启用搜索功能，值为：true/fals,默认false isSearch:true
callback 选中下拉时的回调函数 返回 选中的 name和id callback: function(name,val) {alert(name);alert(val);}
